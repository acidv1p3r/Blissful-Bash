# Function to replace apt-get with apt, using Nala where applicable
apt-get() {
    # Check if the command is being called with sudo
    if [[ "$EUID" -eq 0 ]]; then
        echo "Intercepted (sudo): $1"
    else
        echo "Intercepted: $1"
    fi

    # Determine which apt-get command is being run
    case "$1" in
        update)
            echo "Running: nala update"
            sudo nala update || echo "Nala update failed."
            ;;
        upgrade)
            echo "Running: nala upgrade -y"
            sudo nala upgrade -y || echo "Nala upgrade failed."
            ;;
        install|remove|autoremove)
            echo "Running: sudo nala $1 ${@:2}"
            sudo nala "$1" "${@:2}" || echo "Nala $1 failed."
            ;;
        *)
            echo "Running original apt-get command"
            command apt-get "$@" || echo "Original apt-get command failed."
            ;;
    esac
}
